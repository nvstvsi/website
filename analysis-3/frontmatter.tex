% ============================================================================
% frontmatter.tex
% Configuration for document frontmatter styling and layout
% ============================================================================

% ============================================================================
% Page Style & Headers/Footers
% ============================================================================
\usepackage{fancyhdr}
\pagestyle{fancy}

\fancyhf{}  % Clear default headers/footers
\fancyhead[LE]{\textsl{\thepage}}
\fancyhead[RO]{\textsl{\thepage}}
\fancyhead[RE]{\textsl{\leftmark}}
\fancyhead[LO]{\textsl{\rightmark}}
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\footrulewidth}{0pt}

% Plain style for chapter starts and front matter
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% ============================================================================
% Title Page Customization
% ============================================================================
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \centering
    \vspace*{2cm}
    
    {\LARGE \textbf{\@title} \par}
    
    \vspace{2cm}
    
    {\large \@author \par}
    
    \vfill
    
    {\large \@date \par}
    
    \vspace{2cm}
  \end{titlepage}
  \thispagestyle{empty}
  \newpage
}
\makeatother

% ============================================================================
% Table of Contents Customization
% ============================================================================
\usepackage{tocloft}

% Reduce spacing in TOC
\setlength{\cftbeforepartskip}{0pt}
\setlength{\cftbeforechapskip}{6pt plus 2pt}
\setlength{\cftbeforesecskip}{3pt}

% Make chapter numbers in TOC bold
\renewcommand{\cftchapfont}{\bfseries}
\renewcommand{\cftchappagefont}{\bfseries}

% ============================================================================
% Chapter Styling
% ============================================================================
\usepackage{titlesec}

% Chapter title format
\titleformat{\chapter}[display]
  {\normalfont\Large\bfseries}
  {\chaptertitlename\ \thechapter}
  {20pt}
  {\Huge}
  
\titlespacing*{\chapter}{0pt}{-30pt}{20pt}

% Section and subsection spacing
\titlespacing*{\section}{0pt}{18pt plus 4pt minus 2pt}{12pt plus 2pt minus 2pt}
\titlespacing*{\subsection}{0pt}{12pt plus 3pt minus 1pt}{8pt plus 1.5pt minus 1pt}

% ============================================================================
% Enumerate & List Customization
% ============================================================================
\usepackage{enumitem}

% Reduce spacing in lists
\setlist[enumerate]{topsep=6pt,itemsep=3pt}
\setlist[itemize]{topsep=6pt,itemsep=3pt}

% ============================================================================
% Hyperref Configuration
% ============================================================================
\hypersetup{
  colorlinks=true,
  linkcolor=black,
  citecolor=black,
  urlcolor=blue,
  bookmarksopen=false,
  pdfpagemode=UseNone,
  pdfstartpage=1
}

% ============================================================================
% Micro-typography
% ============================================================================
\usepackage[protrusion=true,expansion=true]{microtype}

% Increase line spacing slightly for readability
% (already set via \linespread{1.2} in preamble)

% ============================================================================
% Section Numbering Depth
% ============================================================================
\setcounter{secnumdepth}{2}  % Number chapters, sections, subsections
\setcounter{tocdepth}{2}      % TOC shows chapters, sections, subsections
